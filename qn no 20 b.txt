using System;
using System.Collections.Generic;

class Program
{
    static void Main()
    {
        Stack<int> stack = new Stack<int>();

        stack.Push(10);
        stack.Push(20);
        stack.Push(30);

        Console.WriteLine("Pop: " + stack.Pop());

        foreach (int item in stack)
        {
            Console.WriteLine(item);
        }
    }
}4
