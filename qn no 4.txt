using System;

class Student
{
    int id;
    string name;

    // Default Constructor
    public Student()
    {
        id = 0;
        name = "Unknown";
    }

    // Parameterized Constructor
    public Student(int i, string n)
    {
        id = i;
        name = n;
    }

    // Copy Constructor
    public Student(Student s)
    {
        id = s.id;
        name = s.name;
    }

    public void Display()
    {
        Console.WriteLine("ID: " + id + " Name: " + name);
    }
}

class Program
{
    static void Main()
    {
        Student s1 = new Student();
        Student s2 = new Student(101, "Ram");
        Student s3 = new Student(s2);

        s1.Display();
        s2.Display();
        s3.Display();
    }
}
