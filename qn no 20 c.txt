using System;
using System.Collections.Generic;

class Program
{
    static void Main()
    {
        Queue<string> queue = new Queue<string>();

        queue.Enqueue("Ram");
        queue.Enqueue("Shyam");
        queue.Enqueue("Hari");

        Console.WriteLine("Removed: " + queue.Dequeue());

        foreach (string item in queue)
        {
            Console.WriteLine(item);
        }
    }
}
